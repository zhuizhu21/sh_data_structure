```c++
bool Del_same(SqList &L){
    if(L.length<=1){
        return true;
    }
    ElemType temp=L.data[0];
    int i=1,j=0;
    for(i=1;i<L.length;i++){
        if(L.data[i]!=temp){
            L.data[j]=L.data[i];
            temp=L.data[i];
            j++;
        }
    }
    L.length=j;
    return true;
}
```
